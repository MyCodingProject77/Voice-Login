<script>
let usernameValue = "";
let passwordValue = "";
let recognition;
let lastTranscript = "";
let silenceTimer;

function normalizePassword(text) {
    text = text.toLowerCase();

    // Handle common mis-hearings of "cool"
    if (text.includes("cool") || text.includes("kool") || text.includes("call") || text.includes("coo")) {
        return "cool";
    }
    return text;
}

function startVoice(type) {
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

    if (!SpeechRecognition) {
        alert("Use Google Chrome");
        return;
    }

    if (recognition) {
        recognition.abort();
    }

    recognition = new SpeechRecognition();
    recognition.lang = "en-US";
    recognition.interimResults = true;
    recognition.continuous = false;

    lastTranscript = "";

    document.getElementById("message").innerText = "🎤 Listening...";

    recognition.start();

    recognition.onresult = function(event) {
        let text = event.results[0][0].transcript.toLowerCase().trim();
        lastTranscript = text;

        console.log("Heard:", text);

        // Reset silence timer every time speech is detected
        clearTimeout(silenceTimer);
        silenceTimer = setTimeout(() => {
            processResult(type, lastTranscript);
            recognition.stop();
        }, 1200); // 1.2 sec silence = done speaking
    };

    recognition.onerror = function(event) {
        document.getElementById("message").innerText = "❌ Error: " + event.error;
    };

    recognition.onend = function() {
        console.log("Stopped");
    };
}

function processResult(type, text) {
    if (!text) {
        document.getElementById("message").innerText = "❌ Didn't catch that. Try again.";
        return;
    }

    if (type === "username") {
        usernameValue = text;
        document.getElementById("message").innerText = "✅ Username: " + text;
    } else {
        passwordValue = normalizePassword(text);
        document.getElementById("message").innerText = "✅ Password: " + passwordValue;
    }
}

function login() {
    let user = usernameValue;
    let pass = passwordValue;

    if (user.includes("shivaji") && user.includes("boss") && pass === "cool") {
        document.getElementById("message").innerText = "🔥 Login Successful! Boss Entry!";
        document.getElementById("message").style.color = "lightgreen";
    } else {
        document.getElementById("message").innerText = "❌ Invalid login!";
        document.getElementById("message").style.color = "red";
    }
}
